Import Data 

talls <- read.csv("~/tallestbuildings.csv")

1.) Buildings Per City 

bpc = talls %>% group_by(City) %>% summarize(Citycount = length(City))

ggplot(bpc, aes(x = reorder(City, Citycount), y = Citycount)) + geom_bar(stat = "identity") + coord_flip() + labs(y = "City Count", x = "Cities", title = "Buildings/City")

2.) City Buildings by Mean Height (Meters) 

cbh = talls %>% group_by(City) %>% summarize(Average = mean(HeightMeter))

ggplot(cbh, aes(x = reorder(City,Average), y = Average)) + geom_bar(stat = "identity") + coord_flip() + labs(y = "Mean Height (Meters)", x = "Cities", title = "City Buildings by Height")

3.) Buildings Per Country 

#First need to separate the variable
talls$City = as.character(talls$City)

#Pull out parens
talls$City = gsub(")","",talls$City)

#Insert _ sep 
talls$City = gsub( "\\(","_",talls$City)

#Now we can take only the codes
bpco = talls %>% separate(City, c("City", "Country"), sep = "_") %>% group_by(Country) %>% summarize(Countrycount = length(Country))

bpco$Country = countrycode(bpco$Country,"iso2c", "country.name")

ggplot(bpco, aes(x = reorder(Country, Countrycount), y = Countrycount)) + geom_bar(stat = "identity") + coord_flip() + labs(y = "Country Count", x = "Countries", title = "Buildings/Country")

4.) Country Buildings by Mean Height (Meters) 

cubh = talls %>% separate(City, c("City", "Country"), sep = "_") %>% group_by(Country) %>% summarize(CAverage = mean(HeightMeter))

cubh$Country = countrycode(cubh$Country,"iso2c", "country.name")

ggplot(cubh, aes(x = reorder(Country,CAverage), y = CAverage)) + geom_bar(stat = "identity") + coord_flip() + labs(y = "Mean Height (Meters)", x = "Countries", title = "Country Buildings by Height")

5.) Colored Country Buildings by Mean Height (Meters) 

ggplot(cubh, aes(x = reorder(Country,CAverage), y = CAverage, fill = Country)) + geom_bar(stat = "identity") + coord_flip() + labs(y = "Mean Height (Meters)", x = "Countries", title = "Country Buildings by Height")
